\section{Find the derivatives with respect to components of the independent variable with the bases assumed constant.}
\begin{enumerate}[label = (\roman*)]
% ---- i -----
	\item Show that for $\phi = (\bm{v} \cdot \bm{v} )^{3/2} ; 
		\qquad (\phi) \overset{\leftharpoonup}{ \nabla}_{\bm{v}}=
		\frac{3 }{2}(\bm{v} \cdot \bm{v} )^{1/2} \bm{v}$
			
		\begin{itemize}
		\item $\phi = (v_j v_j)^{3/2}$
		\item $(\phi) \overset{\leftharpoonup}{ \nabla}_{\bm{v}} 
			= \frac{\partial \phi}{\partial \bm{v}} 
			\Rightarrow \frac{d \phi}{d v_i} = \frac{3}{2}(v_j v_j)^{1/2}  \frac{d(v_k v_k)}{dv_i}$
		\item$ \frac{d(v_k v_k)}{dv_i} = 2 v_l \frac{\partial v_l}{\partial v_i} = 2 v_l \delta_{il} = 2v_l$
		\item $(\phi) \overset{\leftharpoonup}{ \nabla}_{\bm{v}} =
				\frac{3}{2}(v_j v_j)^{1/2} 2v_l = 3(v_j v_j)^{1/2}v_l =
				 3(\bm{v} \cdot \bm{v})^{1/2}\bm{v}$
	\end{itemize}
				
% ---- ii ----
	\item Show that for $\phi = tr(\bm{T}) + tr(\bm{T}^2) + (\bm{T} \cdot \cdot \bm{T})^{1/2}; \qquad
		(\phi) \overset{\leftharpoonup}{\nabla}_{\bm{T}} = \bm{I} + 2 \bm{I}^T + \frac{\bm{T}}{(\bm{T} \cdot \cdot \bm{T})^{1/2}}$
	\begin{itemize}
		\item$\phi= T_{pp} + T_{pq}T_{qp} + (T_{pq}T_{qp})^{1/2}$
		
		\item \begin{flalign*} (\phi) \overset{\leftharpoonup}{\nabla}_{\bm{T}} &= \frac{\partial \phi}{\partial \bm{T}} &\\
			 & = \frac{\partial T_{pp}}{ \partial T_{ij}} + \left(T_{pq} \frac{T_{qp}}{\partial T_{ij}} +
				 T_{qp} \frac{T_{pq}}{\partial T_{ij}} \right) + \frac{1}{2} \left( T_{pq} T_{qp} \right)^{-1/2} 
			 	\frac{\partial (T_{rs}T_{sr})}{\partial T_{ij}}\\
			& = \delta_{pi} \delta_{pj} + 2T_{pq} \delta_{pi} \delta_{qj} + \frac{1}{2(T_{pq} T_{qp})^{1/2}}
				2T_{rs}\delta_{ri} \delta_{sj}\\
			& = \delta_{ij} + 2T_{ij} + \frac{T_{ij}}{(T_{pq}T_{pq})^{1/2}} \Rightarrow
			\bm{I} + 2 \bm{T}^T + \frac{\bm{T}}{(\bm{T} \cdot \cdot \bm{T})^{1/2}}
		\end{flalign*}
	\end{itemize}
	
% ---- iii ----
	\item  Show that for $\phi = \bm{v}(\bm{v} \cdot \bm{v})^{3/2}; \qquad
			(\bm{\phi})\overset{\leftharpoonup}{\nabla}_{\bm{v}} = \Big[ \bm{I}(\bm{v} \cdot \bm{v})^{3/2} + 3 (\bm{v}
			\cdot \bm{v})^{1/2} (\bm{v} \otimes \bm{v})\Big]$
	\begin{itemize}
		\item Postulate: $(\bm{\phi})\overset{\leftharpoonup}{\nabla}_{\bm{v}} =
			\frac{\partial \bm{\phi}}{\partial v_i} \otimes  \bm{e}_i$
		\item $\bm{\phi} = v_k \bm{e}_k(v_l v_l)^{3/2}$
		\item \begin{flalign*}
			\frac{\partial \bm{\phi}}{\partial v_i} &= \frac{\partial v_k}{\partial v_i} \bm{e}_k(v_l v_l)^{3/2} + 
			v_k \bm{e}_k \frac{\partial(v_l v_l)}{\partial v_i}&\\
			&= \delta_{ki} \bm{e}_k (v_l v_l)^{3/2} + v_k \bm{e}_k (3/2)(v_l v_l)^{1/2} 
			\left( v_m \frac{\partial v_m}{\partial v_i} + \frac{\partial v_m}{\partial v_i} v_m \right) \\
			&= \delta_{ki} \bm{e}_k (v_l v_l)^{3/2} + v_k \bm{e}_k (3/2)(v_l v_l)^{1/2} (2v_m \delta_{mi})\\
			&=(\bm{v} \cdot \bm{v})^{3/2} \bm{e}_i+ 3 (\bm{v} \cdot \bm{v})^{1/2} v_k \bm{e}_k v_i
		\end{flalign*}
		
		\item \begin{flalign*} \frac{\partial \bm{\phi}}{\partial v_i} \otimes \bm{e}_i &=  
			(\bm{v} \cdot \bm{v})^{3/2} \bm{e}_i \otimes \bm{e}_i+ 
				3 (\bm{v} \cdot \bm{v})^{1/2} v_k\bm{e}_k v_i \otimes \bm{e}_i&\\
			&=(\bm{v} \cdot \bm{v})^{3/2} \bm{e}_i \otimes \bm{e}_i+ 
				3 (\bm{v} \cdot \bm{v})^{1/2} ( \bm{v} \otimes \bm{v})\\
			&=\bm{I} (\bm{v} \cdot \bm{v})^{3/2} + 
				3 (\bm{v} \cdot \bm{v})^{1/2} ( \bm{v} \otimes \bm{v})\\
		\end{flalign*}
	\end{itemize}	
\end{enumerate}
